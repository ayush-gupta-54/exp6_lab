pipeline { 
    agent any 
    tools { 
        maven 'MAVEN' 
        jdk 'JDK21' 
    } 
    stages { 
        stage('Checkout') { 
            steps { 
                git branch: 'main', url: 'https://github.com/ayush-gupta-54/exp6_lab.git' 
            } 
        } 
        stage('Build') { 
            steps { 
                bat 'mvn clean package' 
            } 
        } 
        stage('Test') { 
            steps { 
                bat 'mvn test' 
            } 
        } 
    } 
    post { 
        always { 
            junit allowEmptyResults: true, testResults: '**/target/surefire-reports/*.xml' 
        } 
    } 
}
